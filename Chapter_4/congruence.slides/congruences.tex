\slide{
	\begin{defn}
		Let $n$ be a positive integer. We will say the integers $a$ and $b$ are \emph{congruent modulo $n$} if $n$ divides $a-b$.  We denote this by
			\[a\equiv b\pmod n.\]
	\end{defn}
	\begin{exa}
		\bulletize{
			\item $n=7$, $a=73$, $b=52$\vskip .5in
			\item $n=23$, $a=-5$ what $b$'s will work?
		}
	\end{exa}
}
\slide{
	\begin{question}
		What happens if $n=1$?
	\end{question}
	\vfill
	\begin{question}
		What numbers are congruent when $n=2$?
	\end{question}
	\vfill
}
\slide{
	\begin{thm}{4.1}
		For arbitrary integers $a$ and $b$, $a\equiv b\pmod n$ if and only if $a$ and $b$ leave the same remainder when divided by $n$.
	\end{thm}
}
\slide{
	\begin{statementblock}{Fact}
		If $a\equiv b \pmod n$, then $a^m\equiv b^m\pmod n$ for all $m\in\N$.
	\end{statementblock}
}

\slide{
	\begin{exercise}
		What is the remainder when dividing $9^{10}$ by $7$?
	\end{exercise}
}
\slide{
	\begin{exercise}
		What is the remainder when dividing $2^{100}$ by $11$?
	\end{exercise}
}
\slide{
	\begin{exercise}
		Use congruence theory to show that for all $n\geq 1$
		\[7|5^{2n}+3\cdot 2^{5n-2}\]
	\end{exercise}
}
\slide{

		If $2x\equiv 0\pmod{12}$ what are the possible values of $x$? (We'll just start with $0\leq x< 12$.)
	\end{exercise}

\slide{
	\begin{thm}{4.5}
		Let $N=a_m10^m+a_{m-1}10^{m-1}+\cdots+a_110+a_0$ be the decimal expansion of $N$.  (That is $a_m,a_{m-1},\dots,a_1,a_0$ are the digits of $N$.)  
		
		If $S=a_m+a_{m-1}+\cdots+a_1+a_0$, then $9|N$ if and only if $9|S$.
	\end{thm}
}
\slide{
	\begin{thm}{4.6}
		Let $N=a_m10^m+a_{m-1}10^{m-1}+\cdots+a_110+a_0$ be the decimal expansion of $N$.  (That is $a_m,a_{m-1},\dots,a_1,a_0$ are the digits of $N$.)  
		
		If $T=a_0-a_{1}+a_2-\cdots+(-1)^ma_m$, then $11|N$ if and only if $11|T$.
	\end{thm}
}
